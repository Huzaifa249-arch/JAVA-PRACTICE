2. Small, Large, Equal 
import java.util.Scanner;
public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int a = sc.nextInt();
        int b = sc.nextInt();
        if (a == b) {
            System.out.println("a == b");
        }
        if (a < b) {
            System.out.println("a < b");
        }
        if (a > b) {
            System.out.println("a > b");
        }
    }
}

8. N-TV Discount
import java.util.Scanner;
public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int T = sc.nextInt();
        while (T-- > 0) {
            int A = sc.nextInt();
            int B = sc.nextInt();
            int C = sc.nextInt();
            int D = sc.nextInt();
            int priceFirst = A - C;
            int priceSecond = B - D;            
            if (priceFirst < priceSecond) {
                System.out.println("First");
            } else if (priceSecond < priceFirst) {
                System.out.println("Second");
            } else {
                System.out.println("Any");
            }
        }
    }
}

10. P- Chef and Candies
import java.util.Scanner;
public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int T = sc.nextInt();
        while (T-- > 0) {
            int N = sc.nextInt();
            int X = sc.nextInt();
            if (X >= N) {
                System.out.println(0);
            } else {
                int needed = N - X;
                int packets = (needed + 3) / 4;
                System.out.println(packets);
            }
        }
    }
}
